SELECT * FROM tb_equipamento;

SELECT * FROM tb_movimentacao;

SELECT * FROM tb_colaborador;


SELECT e.modelo, e.identificador, e.linha, f.nome AS nome_colaborador, f.funcao, o.tipo_movimentacao
FROM tb_equipamento e
INNER JOIN tb_movimentacao o ON e.id = o.id_equipamento
INNER JOIN tb_colaborador f ON o.id_colaborador = f.id
WHERE e.tipo = 'Telefone' AND o.tipo_movimentacao = 'SAIDA' AND e.emprestimo = true;
